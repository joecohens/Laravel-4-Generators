<?php

class OrdersControllerTest extends TestCase {

    /** @test **/
    public function it_should_get_index()
    {
        $this->call('GET', '/orders');

        $this->assertResponseOk();
    }

    /** @test **/
    public function it_should_get_new()
    {
        $this->call('GET', '/orders/create');

        $this->assertResponseOk();
    }

    /** @test **/
    public function it_should_create_order()
    {
        $this->call('POST', '/orders');

        $this->assertResponseOk();
    }

    /** @test **/
    public function it_should_show_order()
    {
        $this->call('GET', '/orders', array('order' => 1));

        $this->assertResponseOk();
    }

    /** @test **/
    public function it_should_get_edit()
    {
        $this->call('GET', '/orders/1/edit');

        $this->assertResponseOk();
    }

    /** @test **/
    public function it_should_update_order()
    {
        $this->call('PUT', '/orders/1');

        $this->assertResponseOk();
    }

    /** @test **/
    public function it_should_destroy_order()
    {
        $this->call('DELETE', '/orders/1');

        $this->assertResponseOk();
    }

}
